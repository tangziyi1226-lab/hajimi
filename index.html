统一动作接口

import createCat from './cats/orange-tabby/cat.js';

const cat = createCat();
cat.mount('#sprite');

// 方式一：方法调用
cat.forward();    // 前倾
cat.happy();      // 开心
cat.tired();      // 疲惫

// 方式二：play 通用接口
cat.play('healthy');
cat.play('tiltLeft');

// 获取所有状态
cat.getStates();  // [{ key, label, frames }, ...]
每个品种独立文件夹，可直接 import createCat from './cats/xxx/cat.js' 复用
